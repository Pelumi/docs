# `POST /v1/direct-debits`

#### Request body


<ParamField body="customerName" type="string" required>
  The customer's account name.
</ParamField>

<ParamField body="customerPhoneNumber" type="string" required>
  The phone of the customer.
</ParamField>

<ParamField body="customerEmail" type="string" required>
  The email of the customer.
</ParamField>

<ParamField body="customerAddress" type="string" required>
  The address of the customer.
</ParamField>

<ParamField body="amount" type="number" required >
  The amount to be used for the mandate.
</ParamField>

<ParamField body="customerAccountNumber" type="string" required>
  The customer's account number.
</ParamField>

<ParamField body="customerAccountName" type="string" required>
  The customer's account name.
</ParamField>

<ParamField body="bankCode" type="string" required>
  The customer account bank code.
</ParamField>

<ParamField body="merchantReference" type="string" required>
  Unique reference used to track a transaction from an external process.
</ParamField>

<ParamField body="frequency" type="string" required>
  The frequency of the mandate
</ParamField>

<ParamField body="narration" type="string" required>
  The narration for this transfer (NB: This will be appended to the normal system generated narration).
</ParamField>

<ParamField body="startDate" type="date" required>
  The start date for the mandate to commence. This is a future date to the date of creation.
</ParamField>

<ParamField body="endDate" type="date" required>
  The end date for the mandate to stop
</ParamField>

<ParamField body="startImmediately" type="boolean" required>
  This is set to allow the mandate to start running immediately
</ParamField>


#### Response body

<ResponseField name="code" type="string" required>
  Response code
</ResponseField>

<ResponseField name="description" type="string" required>
  Response description
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="object"  defaultOpen="true">
    <ResponseField name="mandateId" type="string" required>
      Mandate ID
    </ResponseField>
    <ResponseField name="merchantReference" type="string" required>
      Merchant Reference
    </ResponseField>
    <ResponseField name="customerPhoneNumber" type="string" required>
      Customer Phone Number
    </ResponseField>
    <ResponseField name="description" type="string" required>
      Description
    </ResponseField> 
  </Expandable>
</ResponseField>

<ResponseField name="message" type="string" required>
  Message
</ResponseField>

<ResponseField name="status" type="boolean" required>
  status
</ResponseField>
